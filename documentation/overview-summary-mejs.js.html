// Dependencies.
var sys = require('sys');
var utils = require('express/utils');

// Helper functions available in templates.
var h = utils.escape;



// This class is created for documentation purposes.
function MEJS() {};

/**
 * Calls the EJS rendering library. Templates are given the MEJS (Mobius Embeded JavaScript)
 * extension to prevent overlap with Express' own EJS paraser.
 *
 * @param {string} content content to render.
 * @param {object} options options for renderer including variables for local scope of template.
 * @type string
 * @return content parsed by the EJS engine.
 * @public
 */
MEJS.prototype.render = function(content, options) {
	// Render the template.
	try {
 		return (new EJS({'text' : content})).render(options['locals']);
	} catch (e) {
		return "An error occured while rendering the template: " + e;
	}
};

exports.render = MEJS.prototype.render;
